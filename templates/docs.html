Basic AI Response: https://ramapi-ixzr.onrender.com/api/ramAI/chat/content/<prompt>

Implementation on JavaScript

function getAIResponse(prompt) {
  fetch(`https://ramapi-ixzr.onrender.com/api/ramAI/chat/content/${encodeURIComponent(prompt)}`)
    .then(response => {
      if (!response.ok) {
        throw new Error("API request failed");
      }
      return response.json();
    })
    .then(data => {
      console.log("AI Response:", data);
      document.getElementById("response").innerText = data;
    })
    .catch(error => {
      console.error("Error:", error);
    });
}

getAIResponse("Hello, how are you?");
